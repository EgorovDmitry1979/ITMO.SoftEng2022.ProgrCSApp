  2namespace CSharp
  3{
  4    /// <summary>
  5    /// Token provides a common abstract base class for 
  6    /// the concrete token implementation classes. It is 
  7    /// intended to be used in conjunction with a specific 
  8    /// token interface. For example, to implement string 
  9    /// literal tokens you would do this:
 10    /// <code>
 11    /// interface IStringLiteralToken : IToken {}
 12    /// class StringLiteralToken : Token, IStringLiteralToken
 13    /// { 
 14    /// ...
 15    /// }
 16    /// </code>
 17    /// </summary>
 18    
 19    internal abstract class Token : IToken
 20    {
 21        /// <param name="begin">begin</param>
 22        /// <param name="end">end</param>
 23        /// <remarks>
 24        /// It would be useful to check if
 25        /// (<paramref name="begin">begin</paramref> ==
 26        /// <paramref name="end">end</paramref>)
 27        /// and throw an exception if so.
 28        /// </remarks>
 29
 30        protected Token(Position begin, Position end)
 31        {
 32            this.begin = begin;
 33            this.end = end;
 34        }
 35    
 36        public override string ToString()
 37        {
 38            return Position.MakeString(begin, end);
 39        }
 40
 41        /// <remarks>
 42        /// Note that Accept is added to the internal
 43        /// implementation hierarchy and not to the 
 44        /// public interface hierarchy. That is, Accept
 45        /// is method of Token but not of IToken.
 46        /// </remarks>
 47        
 48        internal abstract void Accept(ITokenVisitor visitor);
 49                
 50        private readonly Position begin, end;
 51    }
 52}
